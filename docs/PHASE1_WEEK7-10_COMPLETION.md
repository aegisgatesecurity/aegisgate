EXAMPLE USAGE
-------------

# Auto-generate certificates (default)
export AEGISGATE_CERT_DIR=./certs
./aegisgate

# Use external certificates
export AEGISGATE_CERT_FILE=/path/to/cert.pem
export AEGISGATE_KEY_FILE=/path/to/key.pem
./aegisgate

# Mixed mode (auto-generate if external not found)
export AEGISGATE_CERT_FILE=/path/to/cert.pem  # If missing, auto-generate
./aegisgate
